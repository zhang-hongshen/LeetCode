## [32. Longest Valid Parentheses](https://leetcode.com/problems/longest-valid-parentheses/)

### First Solution
**Java**
```java
class Solution {
    public int longestValidParentheses(String s) {
        Stack<Integer> stack = new Stack<>();
        int n = s.length();
        int maxLen = 0;
        stack.push(-1);
        for(int i = 0; i < n; i += 1) {
            if(s.charAt(i) == '(') {
                stack.push(i);
                continue;
            }
            stack.pop();
            if(stack.isEmpty()) {
                stack.push(i);
            }
            maxLen = Math.max(maxLen, i - stack.peek());
        }
        return maxLen;

    }
}
```

**Python**
```python
class Solution:
    def longestValidParentheses(self, s: str) -> int:
        n = len(s)
        stack, max_len = [-1], 0
        for i in range(n):
            if s[i] == '(':
                stack.append(i)
                continue
            stack.pop()
            if not stack:
                stack.append(i)
            max_len = max(max_len, i - stack[-1])
        return max_len
```

### Second Solution
**Java**
```java
class Solution {
    public int longestValidParentheses(String s) {
        int n = s.length();
        // dp[i] represents the length of the longest valid parenthess ends with s[i]
        int[] dp = new int[n];
        int maxLen = 0;
        for(int i = 1; i < n; i += 1) {
            if(s.charAt(i) != ')') {
                continue;
            }
            int pre = i - 1 - dp[i - 1];
            if(pre < 0 || s.charAt(pre) != '(') {
                continue;
            }
            dp[i] = dp[i - 1] + 2;
            if(pre > 0) {
                dp[i] += dp[pre - 1];
            }
            maxLen = Math.max(maxLen, dp[i]);
        }
        return maxLen;
    }
}
```

**Python**
```python
class Solution:
    def longestValidParentheses(self, s: str) -> int:
        n = len(s)
        dp, max_len = [0] * n, 0
        for i in range(1, n):
            if s[i] != ')':
                continue
            pre = i - 1 - dp[i - 1]
            if pre < 0 or s[pre] != '(':
                continue
            dp[i] = dp[i - 1] + 2
            if pre > 0:
                dp[i] += dp[pre - 1]
            max_len = max(max_len, dp[i])
        return max_len
```
