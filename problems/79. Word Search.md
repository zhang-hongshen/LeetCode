## [79. Word Search](https://leetcode.com/problems/word-search/)

### First Solution
**Java**
```java
class Solution {
    public boolean exist(char[][] board, String word) {
        int m = board.length, n = board[0].length;
        for(int i = 0; i < m; i += 1) {
            for(int j = 0; j < n; j += 1) {
                if(dfs(board, word, i, j, 0)) {
                    return true;
                }
            }
        }
        return false;
    }

    private boolean dfs(char[][] board, String word, int row, int col, int k) {
        if(k >= word.length()) {
            return true;
        }
        int m = board.length, n = board[0].length;
        if(row < 0 || row >= m || col < 0 || col >= n 
        || board[row][col] != word.charAt(k)) {
            return false;
        }
        char ch = board[row][col];
        board[row][col] = '*';
        if(dfs(board, word, row - 1, col, k + 1) 
        || dfs(board, word, row + 1, col, k + 1) 
        || dfs(board, word, row, col - 1, k + 1) 
        || dfs(board, word, row, col + 1, k + 1)) {
            return true;
        }
        board[row][col] = ch;
        return false;
    }
}
```

**Python**
```python
class Solution:
    def exist(self, board: List[List[str]], word: str) -> bool:
        m, n = len(board), len(board[0])

        def dfs(row, col, k):
            if k >= len(word):
                return True
            if row < 0 or row >= m or col < 0 or col >= n or board[row][col] != word[k]:
                return False
            temp = board[row][col]
            board[row][col] = "*"
            if dfs(row - 1, col, k + 1) or dfs(row + 1, col, k + 1) or dfs(row, col - 1, k + 1) or dfs(row, col + 1, k + 1):
                return True
            # recover
            board[row][col] = temp
            return False

        for i in range(m):
            for j in range(n):
                if dfs(i, j, 0):
                    return True
        return False
```
