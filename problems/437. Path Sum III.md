## [437. Path Sum III](https://leetcode.com/problems/path-sum-iii/)

### First Solution

**Java**
```java
class Solution {
    public int pathSum(TreeNode root, int targetSum) {
        if(root == null) {
            return 0;
        }
        int fromRoot = pathSumFromRoot(root, targetSum);
        return fromRoot + pathSum(root.left, targetSum) + pathSum(root.right, targetSum);
    }

    private int pathSumFromRoot(TreeNode root, long targetSum) {
        if(root == null) {
            return 0;
        }
        return (root.val == targetSum ? 1 : 0 ) + pathSumFromRoot(root.left, targetSum - root.val) + pathSumFromRoot(root.right, targetSum - root.val);
    }
}
```

class Solution:
    def pathSum(self, root: Optional[TreeNode], target_sum: int) -> int:
        if root is None:
            return 0
        
        def path_sum_from_root(root, target_sum):
            if root is None:
                return 0
            return (1 if root.val == target_sum else 0) + path_sum_from_root(root.left, target_sum - root.val) + path_sum_from_root(root.right, target_sum - root.val)

        from_root = path_sum_from_root(root, target_sum)
        return from_root + self.pathSum(root.left, target_sum) + self.pathSum(root.right, target_sum)
