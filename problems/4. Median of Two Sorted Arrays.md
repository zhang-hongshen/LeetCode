## [4. Median of Two Sorted Arrays](https://leetcode.com/problems/median-of-two-sorted-arrays/)

### First Solution

**Java**
```java
class Solution {
    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        int m = nums1.length, n = nums2.length;
        int[] nums = new int[m + n];
        int i = 0, j = 0, k = 0;
        while(i < m && j < n) {
            if(nums1[i] < nums2[j]) {
                nums[k++] = nums1[i++];
            } else {
                nums[k++] = nums2[j++];
            }
        }
        while(i < m) {
            nums[k++] = nums1[i++];
        }
        while(j < n) {
            nums[k++] = nums2[j++];
        }
        if(k % 2 == 0) {
            return (nums[(k >> 1)- 1] + nums[k >> 1]) / 2.0;
        } 
        return nums[(k - 1) >> 1];
    }
}
```

```java
class Solution {
    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        int m = nums1.length, n = nums2.length;
        if (m > n) {
            return findMedianSortedArrays(nums2, nums1);
        }
        int totalLeft = (m + n + 1) >> 1;
        int l = 0, r = m;
        double res = 0.0;
        while(l <= r) {
            int split1 = (l + r) >> 1;
            int split2 = totalLeft - split1;
            int leftMax1 = split1 > 0 ? nums1[split1 - 1] : Integer.MIN_VALUE;
            int leftMax2 = split2 > 0 ? nums2[split2 - 1] : Integer.MIN_VALUE;
            int rightMin1 = split1 < m ? nums1[split1] : Integer.MAX_VALUE;
            int rightMin2 = split2 < n ? nums2[split2] : Integer.MAX_VALUE;
            if(leftMax1 <= rightMin2 && leftMax2 <= rightMin1) {
                int leftMax = Math.max(leftMax1, leftMax2);
                int rightMin = Math.min(rightMin1, rightMin2);
                res = (m + n) % 2 == 0 ? (leftMax + rightMin) / 2.0 : leftMax;
                break;
            } else if(leftMax1 > rightMin2) {
                r = split1 - 1;
            } else {
                l = split1 + 1;
            }
        }
        return res;
    }
}
```
