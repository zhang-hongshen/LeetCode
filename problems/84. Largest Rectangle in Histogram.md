## [84. Largest Rectangle in Histogram](https://leetcode.cn/problems/largest-rectangle-in-histogram/)

### First Solution
**Java**

```java
class Solution {
    public int largestRectangleArea(int[] heights) {
        int n = heights.length;
        Stack<Integer> stack = new Stack<>();
        int[] left = new int[n];
        Arrays.fill(left, -1);
        for(int i = 0; i < n; i += 1) {
            while(!stack.isEmpty() && heights[i] <= heights[stack.peek()]) {
                stack.pop();
            }
            if(!stack.isEmpty()) {
                left[i] = stack.peek();
            }
            stack.push(i);
        }
        int[] right = new int[n];
        Arrays.fill(right, n);
        stack = new Stack<>();
        for(int i = n - 1; i >= 0; i -= 1) {
            while(!stack.isEmpty() && heights[i] <= heights[stack.peek()]) {
                stack.pop();
            }
            if(!stack.isEmpty()) {
                right[i] = stack.peek();
            }
            stack.push(i);
        }
        int res = 0;
        for(int i = 0; i < n; i += 1) {
            res = Math.max(res, (right[i] - left[i] - 1) * heights[i]);
        }
        return res;
    }
}
```

**Python**
```python
class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        n, stack = len(heights), []
        left = [-1] * n
        for i, height in enumerate(heights):
            # find the first larger element in the left of current element
            while stack and height <= heights[stack[-1]]:    
                stack.pop()
            if stack:
                left[i] = stack[-1]
            stack.append(i)

        right = [n] * n
        stack.clear()
        for i in range(n - 1, -1, -1):
            # find the first larger element in the right of current element
            while stack and heights[i] <= heights[stack[-1]]:    
                stack.pop()
            if stack:
                right[i] = stack[-1]
            stack.append(i)
        res = 0
        for h, l, r in zip(heights, left, right):
            res = max(res, (r - l - 1) * h)
        return res
```
