## [45. Jump Game II](https://leetcode.com/problems/jump-game-ii/)

### First Solution
**Java**
```java
class Solution {
    public int jump(int[] nums) {
        int n = nums.length;
        int[] dp = new int[n];
        Arrays.fill(dp, Integer.MAX_VALUE);
        dp[0] = 0;
        for(int i = 0; i < n; i += 1) {
            if(dp[0] == Integer.MAX_VALUE) {
                continue;
            }
            int limit = Math.min(n, i + nums[i] + 1);
            for(int j = i; j < limit; j += 1) {
                dp[j] = Math.min(dp[j], dp[i] + 1);
            }
        }
        return dp[n - 1];
    }
}
```

**Python**
```python
class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        dp = [inf] * n
        dp[0] = 0
        for i, num in enumerate(nums):
            if dp[i] == inf:
                continue
            limit = min(n, i + num + 1)
            for j in range(i, limit):
                dp[j] = min(dp[j], dp[i] + 1)
        return dp[n - 1]
```

### Second Solution
**Java**
```java
class Solution {
    public int jump(int[] nums) {
        int ans = 0;
        int nearest = 0, farthest = 0;
        int last = nums.length - 1;
        while(farthest < last) {
            int newFarthest = farthest;
            for(int i = nearest; i <= farthest; i += 1) {
                newFarthest = Math.max(newFarthest, i + nums[i]);
            }
            nearest = farthest + 1;
            farthest = newFarthest;
            ans += 1;
        }
        return ans;
    }
}
```

class Solution:

        while farthest < last:
            new_farthest = farthest
            for i in range(nearest, farthest + 1):
                new_farthest = max(new_farthest, i + nums[i])
            nearest = farthest + 1
            farthest = new_farthest
            ans += 1
