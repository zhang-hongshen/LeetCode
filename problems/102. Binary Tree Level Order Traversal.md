## [102. Binary Tree Level Order Traversal](https://leetcode.cn/problems/binary-tree-level-order-traversal/)


### First Solution

**Java**

```java
class Solution {
    public List<List<Integer>> levelOrder(TreeNode root) {
        List<List<Integer>> res = new LinkedList<>();
        if(root == null) {
            return res;
        }
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        while(!q.isEmpty()) {
            int n = q.size();
            List<Integer> level = new LinkedList<>();
            for(int i = 0; i < n; i += 1) {
                TreeNode node = q.poll();
                level.add(node.val);
                if(node.left != null) {
                    q.add(node.left);
                }
                if(node.right != null) {
                    q.add(node.right);
                }
            }
            res.add(level);
        }
        return res;
    }
}
```

```python
        while q:

            for i in range(n):
                node = q.popleft()
                level.append(node.val)
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
            res.append(level)
```

**C++**

```c++
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        if(nullptr == root) {
            return {};
        }
        std::queue<TreeNode*> q;
        vector<vector<int>> res;
        q.push(root);
        while(!q.empty()) {
            int n = q.size();
            vector<int> v;
            v.reserve(n);
            for(int i = 0; i < n; ++i) {
                auto p = q.front();
                q.pop();
                v.push_back(p->val);
                if(nullptr != p->left) {
                    q.push(p->left);
                }
                if(nullptr != p->right) {
                    q.push(p->right);
                }
            }
            res.emplace_back(v);
        }
        return res;
    }
};
```

```typescript
function levelOrder(root: TreeNode | null): number[][] {
    if (root === null) {
        return []
    }
    const res = []
    const q = [root]
    while(q.length > 0) {
        const n = q.length
        const level = []
        for(let i = 0; i < n; i += 1) {
            const node = q.shift()
            level.push(node.val)
            if(node.left !== null) {
                q.push(node.left)
            }
            if(node.right !== null) {
                q.push(node.right)
            }
        }
        res.push(level)
    }
    return res
};
```
